# 写一个方法记录函数运行的时间

function method(callBack) {
  console.time("所用时间");
  callBack();
  console.timeEnd("所用时间");
}

function aa() {
  let num = 0;
  for (let i = 0; i < 5000; i++) {
    num++;
  }
}

method(aa); // 所用时间: 0.10791015625 ms
